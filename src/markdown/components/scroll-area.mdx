  ```
  pnpm dlx shadcn@latest add https://neobrutalism.dev/r/scroll-area.json
  ```
  </TabsContent>
  <TabsContent value="npm">
  ```
  npx shadcn@latest add https://neobrutalism.dev/r/scroll-area.json
  ```
  </TabsContent>
  <TabsContent value="yarn">
  ```
  npx shadcn@latest add https://neobrutalism.dev/r/scroll-area.json
  ```

  </TabsContent>
