
  ```
  npx shadcn@latest add https://neobrutalism.dev/r/input.json
  ```
  </TabsContent>
  <TabsContent value="yarn">
  ```
  npx shadcn@latest add https://neobrutalism.dev/r/input.json
  ```

  <TabsContent value="bun">
